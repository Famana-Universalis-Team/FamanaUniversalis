# Milk Road
country_event = {
	id = prices.1
	title = prices.1.t
	desc = prices.1.d
	picture = TRADEGOODS_eventPicture

	fire_only_once = yes
	major = yes

	trigger = {
		owns_core_province = 5644
		5644 = { #most trade power in Jimkalin
			has_most_province_trade_power = ROOT
		}
		5831 = {
			NOT = { is_empty = yes }
		}
		5826 = {
			NOT = { is_empty = yes }
		}
		is_year = 300
	}

	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = prices.1.a
		change_price = {
			trade_goods = livestock
			key = MILK_ROAD
			value = 0.10
			duration = -1
		}
		change_price = {
			trade_goods = chinaware
			key = MILK_ROAD
			value = 0.10
			duration = -1
		}
		change_price = {
			trade_goods = tea
			key = MILK_ROAD
			value = 0.10
			duration = -1
		}
		add_years_of_income  = 0.25
		add_prestige = 10
		set_global_flag = milk_road_happened
	}
}

# Advancements in Agriculture
country_event = {
	id = prices.3
	title = prices.3.t
	desc = prices.3.d
	picture = FARMING_WIFE_eventPicture

	fire_only_once = yes
	major = yes

	trigger = {
		has_institution = printing_press
		capital_scope = { continent = estal }
		NOT = {
			AND = {
				tag = NAF #nafa is op already as is
				ai = no
			}
		}
	}

	immediate = {
		set_saved_name = {
			key = amazing_naturalist
			type = advisor
			scope = ROOT
		}
	}

	mean_time_to_happen = {
		months = 120
	}
	
	option = {
		name = prices.3.a
		define_advisor = {
			type = natural_scientist
			name = amazing_naturalist
			skill = 3
			discount = yes
		}
		change_price = {
			trade_goods = grain
			key = ALYMIN
			value = 0.25
			duration = 3650
		}
		change_price = {
			trade_goods = tea
			key = ALYMIN
			value = 0.25
			duration = 3650
		}
		change_price = {
			trade_goods = spices
			key = ALYMIN
			value = -0.10
			duration = 3650
		}
		change_price = {
			trade_goods = wine
			key = ALYMIN
			value = -0.10
			duration = 3650
		}
		hidden_effect = {
			every_province = {
				limit = {
					AND = {
						trade_goods = grain
						owner = {
							knows_country = ROOT
						}
					}
				}
				add_province_modifier = {
					name = the_alymin_modifier
					duration = 3650
				}
			}
		}
		custom_tooltip = alyminTooltip
		add_years_of_income  = 0.10
		add_prestige = 10
		set_global_flag = alymin_established
	}
}

# The disappearance of the Ordican Dragon

country_event = {
	id = prices.7
	title = prices.7.t
	desc = prices.7.d
	picture = FEAST_eventPicture

	fire_only_once = yes
	major = yes

	trigger = {
		any_owned_province = {
			trade_goods = ivory
			superregion = west_estal_superregion
		}
		is_year = 466
	}

	mean_time_to_happen = {
		months = 60
	}
	
	option = {
		name = prices.7.a
		change_price = {
			trade_goods = ivory
			key = DISAPPEARANCE_OF_DRAGON
			value = 0.50
			duration = -1
		}
		every_province = {
			limit = {
				trade_goods = ivory
				superregion = west_estal_superregion
			}
			change_trade_goods = unknown
		}
		every_known_country = {
			limit = {
				religion = pawani
			}
			reverse_add_opinion = {
			who = ROOT
			modifier = dragon_killer
			}
		}
	}
}
