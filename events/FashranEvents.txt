country_event = {
    id = fashros.16
    title = fashros.16.t
    picture = BAD_WITH_MONARCH_eventPicture
    desc = fashros.16.d
    fire_only_once = yes
    major = yes

    trigger = {
        AND = {
            is_year = 320
            tag = FAS
            NOT = {
                has_country_modifier = fashros_decay
            }
        }
    }

    mean_time_to_happen = {
        days = 1
    }

    option = {
        name = fashros.16.a
        ai_chance = { factor = 1 }
        add_stability = -6
        kill_ruler = yes
        hidden_effect = {
            #First, cleanse all the existing cores.

            every_subject_country = {
                if = {
                    limit = {
                        is_subject_of = FAS
                    }
                    FAS = {
                        inherit = PREV
                    }
                }
            }

            every_owned_province = {
                limit = {
                    superregion = jasua_superregion
                    superregion = tonmo_supperregion
                }
                every_country = {
                    remove_core = PREV
                }
            }

            #Now, add the cores for the four tonxara nations.
            every_owned_province = {
                if = {
                    limit = {
                        OR = {
                            superregion = west_estal_superregion
                            superregion = ga_xal_superregion
                            superregion = oppia_superregion
                            area = hoxona_area
                            area = hoxoi_area
                            area = ropinver_area
                            area = diam_area
                            area = unnamed_area_11
                            area = mahansha_area
                        }
                    }
                    add_core = JFX
                }
                else_if = {
                    limit = {
                        OR = {
                            superregion = jasua_superregion
                            superregion = central_estal_supperregion
                            superregion = great_desert_superregion
                            superregion = north_estal_superregion
                        }
                    }
                    add_core = KSL
                }
                else = {
                    add_core = NHM
                } #There is also a fourth Tonxara for Geopyopia called the Hoyros Tonxara, but they dont have a superregion yet
                if = {
                    limit = {
                        OR = {
                            area = ufulk_area
                            area = unnamed_area_37
                            area = unnamed_area_36
                            area = yenis_area
                        }
                    }
                    remove_core = KSL
                    add_core = NHM
                }
            }

            #name the country depending on the number of tonxaras
            set_variable = {
                which = tonxaraCount
                value = 0
            }
