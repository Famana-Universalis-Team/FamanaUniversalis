#Jimbalak migration
country_event = {
	id = nhoma.5
	title = nhoma.5.t
    desc = nhoma.5.d
	picture = LAND_MILITARY_eventPicture
	fire_only_once = yes
	hidden = yes
	
	trigger = {
		is_year = 340
		NOT = {
			exists = JIB
		}
        any_owned_province = {
            region = nozumol_region
        }
	}
	
	mean_time_to_happen = {
	    months = 12
	}
	
	immediate = {
        if = {
            limit = {
                6010 = {
                    is_empty = yes
                }
            }
            cede_province = JIB
            add_core = JIB
            change_culture = Jimbalan
            change_religion = tonmo
        }
        if = {
            limit = {
                6567 = {
                    is_empty = yes
                }
            }
            cede_province = JIB
            add_core = JIB
            change_culture = Jimbalan
            change_religion = tonmo
        }
        if = {
            limit = {
                6552 = {
                    is_empty = yes
                }
            }
            cede_province = JIB
            add_core = JIB
            change_culture = Jimbalan
            change_religion = tonmo
        }
        JIB = { change_religion = tonmo }
	}
	option = {
	    name = nhoma.5.a
	}
}

#Khim-fisbiklu alliance
country_event = {
    id = nhoma.6
	title = nhoma.6.t
    desc = nhoma.6.d
	picture = DIPLOMACY_eventPicture
	fire_only_once = yes

    trigger = {
        tag = FSB
        exists = KMI
        NOT = {
            OR = {
                alliance_with = KMI
                is_rival = KMI
                is_enemy = KMI
            }
        }
    }

    immediate = {
        add_historical_friend = KMI
        KMI = {
            add_historical_friend = FSB
        }
    }

    mean_time_to_happen = {
        days = 10
    }

    option = { #Yes
        name = nhoma.6.a
        ai_chance = { factor = 1 }
        KMI = {
            country_event = {
                id = nhoma.7
                days = 1
            }
        }
    }

    option = { #No
        name = nhoma.6.b
        ai_chance = { factor = 0 }
    }
}
